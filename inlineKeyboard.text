def get_agreement(message):
    message_buttons = types.InlineKeyboardMarkup()

    request_yes = types.InlineKeyboardButton("✅", callback_data='1')
    request_no = types.InlineKeyboardButton("⛔️", callback_data='2')

    message_buttons.add(request_yes, request_no)

    bot.send_message(message.chat.id, "Do you request weather information?", reply_markup=message_buttons)

@bot.callback_query_handler(func = lambda call: True)
def get_city(call):
    if call.data == '1':
        message_from_bot = bot.send_message(call.message.chat.id, "🏛 Write the name of сity: ")
        bot.register_next_step_handler(message_from_bot, get_weather)
    elif call.data == '2':

        bot.send_message(call.message.chat.id, "Process finished!")
